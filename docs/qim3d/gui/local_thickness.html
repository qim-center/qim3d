325    def process_input(self, session):
326        # Load volume
327        session.vol = DataLoader().load(session.data.name)
328
329        if session.dark_objects:
330            session.vol = np.invert(session.vol)
331
332        if session.flip_z:
333            session.vol = np.flip(session.vol, axis=0)
334
335        return session
336
337    def prepare_volume(self, session):
338        # Get volume shape
339        session.get_vol_info()
340
341        # Resize for display
342        session.vol_input_display = ndimage.zoom(
343            input=session.vol,
344            zoom=(session.display_size_input / session.max_size),
345            prefilter=False,
346            order=0,
347        )
348
349        display_shape = np.shape(session.vol_input_display)
350        session.input_display_size_z = display_shape[0]
351        session.input_display_size_y = display_shape[1]
352        session.input_display_size_x = display_shape[2]
353
354        # Create 3D grid
355        session.Zgrid, session.Ygrid, session.Xgrid = np.mgrid[
356            0 : session.input_display_size_z,
357            0 : session.input_display_size_y,
358            0 : session.input_display_size_x,
359        ]
360
361        return session

519    def thickness_histogram(self, session):
520        # Ignore zero thickness
521        non_zero_values = session.vol_thickness[session.vol_thickness > 0]
522
523        # Calculate histogram
524        vol_hist, bin_edges = np.histogram(non_zero_values, session.nbins)
525
526        fig, ax = plt.subplots(figsize=(6, 4))
527
528        ax.bar(
529            bin_edges[:-1], vol_hist, width=np.diff(bin_edges), ec="white", align="edge"
530        )
531
532        # Adjustments
533        ax.spines["right"].set_visible(False)
534        ax.spines["top"].set_visible(False)
535        ax.spines["left"].set_visible(True)
536        ax.spines["bottom"].set_visible(True)
537        ax.set_yscale("log")
538
539        return fig
